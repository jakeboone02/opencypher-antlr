SinglePartQuery

EBNF

SinglePartQuery = ({ ReadingClause, [SP] }, Return)
                | ({ ReadingClause, [SP] }, UpdatingClause, { [SP], UpdatingClause }, [[SP], Return])
                ;

ReadingClause = Match
              | Unwind
              | InQueryCall
              ;

UpdatingClause = Create
               | Merge
               | Delete
               | Set
               | Remove
               ;
